    <?php
	$data = $this->dataStore('result_0');
    ColumnChart::create(array(
        "title"=>"Sale Report",
        "dataSource"=>$data,
        "columns"=>array(
            "category",
            "sale"=>array("label"=>"Sale","type"=>"number","prefix"=>"$"),
            "cost"=>array("label"=>"Cost","type"=>"number","prefix"=>"$"),

        ),
        "colorScheme"=>array(
            "#e7717d",

        )
    ));

        "title"=>"Sale Report on Stack",

        )
    ));
